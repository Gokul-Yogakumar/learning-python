a=4
b=6

#Addition
print('a + b =',a+b)

#Subtraction
print('a - b =',a-b)

#Multiplication
print('a * b =',a*b)

#Division
print('a / b =',a/b)

#Modulo division, prints the remainder
print('a % b=',a%b)

#Floor division
print('a // b =',a//b)

#Exponent a to the power b
print('a ** b =',a**b)
